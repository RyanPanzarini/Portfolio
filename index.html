select substr(sus_obter_dados_equipe(b.NR_SEQ_SUS_EQUIPE,'NM'),1,100)equipe,
obter_cpf_pessoa_fisica(a.CD_PESSOA_FIS_RESPON)cpf_resp,
obter_cns(a.CD_PESSOA_FIS_RESPON)cns_resp,
a.CD_PESSOA_FIS_RESPON cod_pessoa,
obter_nome_pf(a.CD_PESSOA_FIS_RESPON)pessoa_resp,
to_char(obter_data_nascto_pf(a.CD_PESSOA_FIS_RESPON),'dd/mm/yyyy')nasc_resp,
obter_sexo_pf(a.CD_PESSOA_FIS_RESPON,'C')sexo_resp,

obter_cpf_pessoa_fisica(b.CD_PESSOA_FISICA)cpf_dep,
obter_cns(b.cd_pessoa_fisica)cns,
obter_nome_pf(b.CD_PESSOA_FISICA)pessoa_dep,
c.ds_logradouro casa,
to_char(obter_data_nascto_pf(b.CD_PESSOA_FISICA),'dd/mm/yyyy')nasc_dep,
obter_sexo_pf(b.CD_PESSOA_FISICA,'C')sexo_dep,
substr(obter_desc_micro_area(b.NR_SEQ_MICRO_AREA),1,255) microarea,
obter_nome_estabelecimento(b.cd_estabelecimento)estabelecimento

from ESUS_CAD_DOM_FAMILIA a,
ESUS_CAD_INDIVIDUAL b,
ESUS_CAD_DOMICILIAR c

where  a.cd_pessoa_fis_respon = b.CD_PESSOA_RESPONSAVEL
and a.nr_seq_cad_dom = c.nr_sequencia
and c.dt_liberacao is not null
and b.dt_liberacao is not null
and b.dt_liberacao BETWEEN inicio_dia(:dt_inicial) and fim_dia(:dt_final)
and (b.CD_PROFISSIONAL = :CD_PROFISSIONAL or (:CD_PROFISSIONAL = '0'))
and ((b.cd_estabelecimento = :cd_estab) OR (:cd_estab = '0'))
and ((b.NR_SEQ_SUS_EQUIPE = :nr_equipe) or (:nr_equipe = '0'))

and c.nr_sequencia = (
select max (x.nr_sequencia)
from ESUS_CAD_DOMICILIAR x
where x.cd_cep = c.cd_cep
and x.NR_LOGRADOURO = c.nr_logradouro
and c.dt_liberacao is not null
)
order by casa, pessoa_resp

Relatórios 

-- Relatório de procedimentos por especialidade
SELECT 
    e.ds_especialidade,
    COUNT(p.cd_procedimento) AS total_procedimentos,
    SUM(p.vl_procedimento) AS valor_total,
    AVG(p.vl_procedimento) AS valor_medio
FROM 
    procedimento p
JOIN 
    especialidade e ON p.cd_especialidade = e.cd_especialidade
WHERE 
    p.dt_procedimento BETWEEN '2023-01-01' AND '2023-12-31'
GROUP BY 
    e.ds_especialidade
HAVING 
    COUNT(p.cd_procedimento) > 10
ORDER BY 
    valor_total DESC;

-- Inserção de Diu
SELECT PPD.DT_REGISTRO REGISTRO,
OBTER_NOME_PF(PPD.CD_PESSOA_FISICA)PACIENTE,
FLOOR((TO_CHAR(SYSDATE, 'YYYYMMDD') - TO_CHAR(OBTER_DATA_NASCTO_PF(PPD.CD_PESSOA_FISICA), 'YYYYMMDD')) / 10000) AS IDADE,
HSM.QT_GESTACOES GESTAÇÕES,
OBTER_NOME_ESTABELECIMENTO(OBTER_ESTAB_ATEND(PPD.NR_ATENDIMENTO))ESTABELECIMENTO,
OBTER_NOME_PF(PPD.CD_PROFISSIONAL) PROFISSIONAL,
HSM.DS_OBSERVACAO OBSERVACAO




FROM PROC_PAC_DESCRICAO PPD,
HISTORICO_SAUDE_MULHER HSM

WHERE PPD.CD_PESSOA_FISICA = HSM.CD_PESSOA_FISICA
AND PPD.NR_SEQ_PROC_INTERNO = '38612'
AND ((obter_estab_atendimento(PPD.nr_atendimento) = :cd_estab) or (:cd_estab = '0'))
AND PPD.DT_REGISTRO between inicio_dia(:dt_inicial) and fim_dia(:dt_final)
AND ((PPD.CD_PROFISSIONAL = :CD_PROFISSIONAL) OR (:CD_PROFISSIONAL = '0'))
AND :IE_SINTETICO='N'

ORDER BY PPD.NR_ATENDIMENTO ASC
